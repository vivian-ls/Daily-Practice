void solve(){
    int n, l, r;
    cin >> n >> l >> r;
    vector<int> val(n + 1);
    fore(i, 1, n + 1) cin >> val[i];
    int len = max(n - r, l - 1);
    priority_queue<int> pq;
    fore(i, 1, len + 1) pq.push(val[i]);
    int ans = 0;
    fore(i, len + 1, len + r - l + 2) {
        if(val[i] > pq.top())
            ans += val[i];
        else {
            ans += pq.top();
            pq.pop();
            pq.emplace(val[i]);
        }
    }
    cout << ans << endl;
}
