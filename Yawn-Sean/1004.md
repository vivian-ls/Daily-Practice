void solve(){
    int n;
    cin >> n;
    string s;
    cin >> s;
    s = " " + s;
    vector<pii> ans(n + 1);
    int odd = 1, even = 1;
    fore(i, 1, n + 1) {
        if(s[i] == 'O') {
            ans[i] = {odd++ * 2 - 1, even++ * 2};
        }else{
            if(even < odd) ans[i] = {even++ * 2, even++ * 2};
            else ans[i] = {odd++ * 2 - 1, odd++ * 2 - 1};
        }
    }
    int flag = true;
    fore(i, 2, n + 1) {
        int s1 = ans[i].first + ans[i].second;
        int s2 = ans[i - 1].first + ans[i - 1].second;
        if(s2 > s1)
            flag = false;
    }
    fore(i, 1, n + 1) {
        if(ans[i].first > n * 2 || ans[i].second > n * 2)
            flag = false;
    }
    if(flag) {
        cout << "Yes" << endl;
        fore(i, 1, n + 1) cout << ans[i].first << ' ' << ans[i].second << endl;
    }else{
        cout << "No" << endl;
    }
}

void solve(){
    int n, m, k;
    cin >> n >> m >> k;
    vector<int> a(k);
    fore(i, 0, k) cin >> a[i];
    vector<vector<int>> p(n + 1);
    int u, v;
    fore(i, 1, m + 1) {
        cin >> u >> v;
        p[u].emplace_back(v);
        p[v].emplace_back(u);
    }
    int cur = 1;
    for(int x : a) {
        if(x != cur) {
            p[x].emplace_back(cur);
            p[cur].emplace_back(x);
        }
        cur = x;
    }
    vector<int> dis(n + 1, -1);
    dis[1] = 0;
    queue<int> q;
    q.emplace(1);
    while(!q.empty()) {
        int s = q.front(); q.pop();
        for(int to : p[s]) {
            if(dis[to] == -1) {
                dis[to] = dis[s] + 1;
                q.emplace(to);
            }
        }
    }
    int flag = true;
    fore(i, 1, k) {
        if(dis[a[i]] <= dis[a[i - 1]])
            flag = false;
    }
    if(flag) {
        cout << (a[0] == 1 ? k - 1 : k) << endl;
        int cur = 1;
        for(int x : a) {
            if(cur != x) cout << cur << ' ' << x << endl;
            cur = x;
        }
    }else{
        cout << -1 << endl;
    }
}
